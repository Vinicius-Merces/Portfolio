// Controle de Agendamentos class AgendamentoManager { constructor() { this.form = document.getElementById('agendamentoForm'); this.initEventListeners(); } initEventListeners() { this.form.addEventListener('submit', async (e) => await this.handleSubmit(e)); } async handleSubmit(e) { e.preventDefault(); const btn = this.form.querySelector('button[type="submit"]'); btn.disabled = true; document.getElementById('loadingSpinner').style.display = 'inline-block'; try { const agendamento = { nome: document.getElementById('nomeCompleto').value.trim(), whatsapp: document.getElementById('whatsapp').value.trim(), servico: document.getElementById('servico').value, dataHora: new Date(document.getElementById('dataHora').value), observacoes: document.getElementById('observacoes').value.trim(), status: 'pendente', timestamp: firebase.firestore.FieldValue.serverTimestamp() }; // Validar dados this.validarAgendamento(agendamento); // Salvar no Firestore const docRef = await db.collection('agendamentos').add(agendamento); // Enviar para WhatsApp await this.enviarWhatsApp(agendamento, docRef.id); // Feedback ao usuário this.showMessage('Agendamento confirmado! Verifique seu WhatsApp.', 'success'); this.form.reset(); } catch (error) { this.showMessage(`Erro: ${error.message}`, 'danger'); } finally { btn.disabled = false; document.getElementById('loadingSpinner').style.display = 'none'; } } validarAgendamento(agendamento) { const horaMinima = new Date(); horaMinima.setHours(horaMinima.getHours() + 2); if (agendamento.dataHora < horaMinima) { throw new Error('Agendamento deve ser feito com pelo menos 2 horas de antecedência'); } if (!/^[0-9]{11}$/.test(agendamento.whatsapp)) { throw new Error('WhatsApp deve conter 11 números (DDD + número)'); } } async enviarWhatsApp(agendamento, id) { const mensagem = `✅ *Agendamento Confirmado* ✅\n\n` + `*Código:* ${id}\n` + `*Nome:* ${agendamento.nome}\n` + `*Serviço:* ${agendamento.servico}\n` + `*Data/Hora:* ${agendamento.dataHora.toLocaleString('pt-BR')}\n\n` + `_Qualquer alteração, entre em contato conosco!_`; const linkWhatsApp = `https://wa.me/55${agendamento.whatsapp}?text=${encodeURIComponent(mensagem)}`; window.open(linkWhatsApp, '_blank'); } showMessage(message, type) { const div = document.getElementById('statusMessage'); div.innerHTML = `
